# PC调拨超收比例配置表
CREATE TABLE `pc_over_ratio_config` (
    `id` bigint NOT NULL AUTO_INCREMENT COMMENT '主键id',
	`type` tinyint NOT NULL DEFAULT 0 COMMENT '-1-全部, 1-PC超收, 2-3PL超收',
    `management_city_id` bigint NOT NULL DEFAULT 0 COMMENT '经营城市id, -1-全部',
    `sku_id` bigint NOT NULL DEFAULT 0 COMMENT '商品id, -1-全部',
    `priority` int NOT NULL DEFAULT 0 COMMENT '优先级, 数字越小优先级越高',
    `over_ratio` decimal(10, 2) NOT NULL DEFAULT 0.00 COMMENT '超收比例',
    `status` tinyint NOT NULL DEFAULT 0 COMMENT '状态, 0-生效, 1-已删除',
    `creator` varchar(50) NOT NULL DEFAULT '' COMMENT '创建人mis',
    `ctime` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    `modifier` varchar(50) NOT NULL DEFAULT '' COMMENT '更新人mis',
    `utime` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    `valid` tinyint NOT NULL DEFAULT 1 COMMENT '记录是否有效，0-无效, 1-有效',
    PRIMARY KEY (`id`),
    index `idx_city_sku_type` (`management_city_id`,`sku_id`,`type`)
) ENGINE=InnoDB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8mb4 COMMENT='PC调拨超收比例配置表'
